  if (!user.emailVerified) {
    document.getElementById('verify-email-display').textContent = user.email;

    showScreen('login');
  }

// ════════════════════════════════════════
//  FIRESTORE: LOAD USER DATA
// ════════════════════════════════════════
async function loadUserData(user) {
  const docRef = db.collection('users').doc(user.uid);
  const snap   = await docRef.get();

  if (snap.exists) {
    userData = snap.data();
  } else {
    // First login after signup (doc created on signup but just in case)
    userData = buildDefaultProfile(user);
    await docRef.set(userData);
  }

  // Reset steps if new day
  const today = getToday();
  if (userData.stepDate !== today) {
    userData.steps    = 0;
    userData.stepDate = today;
    scheduleSave();
  }
}

// ════════════════════════════════════════
//  FIRESTORE: SAVE (debounced — 1.5 s)
// ════════════════════════════════════════
function scheduleSave() {
  clearTimeout(saveDebounce);
  saveDebounce = setTimeout(() => persistUserData(), 1500);
}

async function persistUserData() {
  if (!currentFBUser || !userData) return;
  try {
    await db.collection('users').doc(currentFBUser.uid).set(userData, { merge: true });
  } catch (err) {
    console.error('Save error:', err);
  }
}

  });
}

function switchTab(tab) {
  // Hide all
  ['login','signup','forgot'].forEach(t => {
    const panel = document.getElementById('form-' + t);
    if (!panel) return;
    if (t === 'forgot') { panel.style.display = 'none'; panel.classList.remove('active'); }
    else { panel.classList.remove('active'); }
  });

  // Show tabs bar only for login/signup
  const tabsEl = document.getElementById('auth-tabs');

  if (tab === 'forgot') {
    tabsEl.style.display = 'none';
    const fp = document.getElementById('form-forgot');
    fp.style.display = 'block';
    fp.classList.add('active');
    clearFormMessages('forgot-error','forgot-success');
  } else {
    tabsEl.style.display = 'flex';
    document.getElementById('form-' + tab).classList.add('active');
    document.querySelectorAll('.tab-btn').forEach((b,i) =>
      b.classList.toggle('active', (i===0 && tab==='login') || (i===1 && tab==='signup'))
    );
    clearFormMessages('login-error','signup-error','signup-success');
  }
}

async function doLogin() {
  clearFormMessages('login-error');
  const email = document.getElementById('login-email').value.trim();
  const pass  = document.getElementById('login-pass').value;

  if (!email || !pass) {
    showFormError('login-error', 'Please fill in both fields.');
    return;
  }

  setLoading('login-btn', true);

  try {
    await auth.signInWithEmailAndPassword(email, pass);
    // onAuthStateChanged handles the rest
  } catch (err) {
    showFormError('login-error', authErrorMsg(err.code));
    setLoading('login-btn', false);
  }
}

// ════════════════════════════════════════
//  AUTH: SIGNUP
// ════════════════════════════════════════
async function doSignup() {
  clearFormMessages('signup-error', 'signup-success');

  const name  = document.getElementById('signup-name').value.trim();
  const email = document.getElementById('signup-email').value.trim();
  const pass  = document.getElementById('signup-pass').value;
  const goal  = document.getElementById('signup-goal').value  || 'general';
  const level = document.getElementById('signup-level').value || 'beginner';

  if (!name)  { showFormError('signup-error', 'Please enter your name.'); return; }
  if (!email) { showFormError('signup-error', 'Please enter your email.'); return; }
  if (pass.length < 6) { showFormError('signup-error', 'Password must be at least 6 characters.'); return; }

  setLoading('signup-btn', true);

  try {
    // 1. Create Firebase Auth user
    const cred = await auth.createUserWithEmailAndPassword(email, pass);
    const user = cred.user;

    // 2. Update display name in Auth
    await user.updateProfile({ displayName: name });

    // 3. Send verification email
    await user.sendEmailVerification();

    // 4. Create Firestore document with signup timestamp + profile
    const profile = {
      name,
      email,
      goal,
      fitnessLevel:     level,
      workouts:         0,
      points:           0,
      streak:           0,
      bestStreak:       0,
      lastWorkoutDate:  null,
      totalReps:        0,
      calories:         0,
      challenges:       0,
      challengeDone:    false,
      challengeDate:    '',
      weeklyData:       [0,0,0,0,0,0,0],
      exerciseCounts:   {},
      steps:            0,
      stepDate:         '',
      reminders:        { workout:false, streak:false },
      signupTimestamp:  firebase.firestore.FieldValue.serverTimestamp(),
      lastLogin:        firebase.firestore.FieldValue.serverTimestamp()
    };

    await db.collection('users').doc(user.uid).set(profile);

    // 5. onAuthStateChanged fires → sees unverified → shows verify screen
    setLoading('signup-btn', false);

  } catch (err) {
    showFormError('signup-error', authErrorMsg(err.code));
    setLoading('signup-btn', false);
  }
}

// ════════════════════════════════════════
//  APP LAUNCH
// ════════════════════════════════════════
function launchApp(user) {
  // Check streak (reset if day was skipped)
  const today    = getToday();
  const yesterday= new Date(Date.now() - 86400000).toISOString().split('T')[0];
  if (userData.lastWorkoutDate && userData.lastWorkoutDate !== today && userData.lastWorkoutDate !== yesterday) {
    userData.streak = 0;
    scheduleSave();
  }

  // Populate nav
  const initials = (userData.name || 'FQ').split(' ').map(n => n[0]).join('').toUpperCase().slice(0,2);
  document.getElementById('nav-avatar').textContent  = initials;
  document.getElementById('nav-username').textContent = (userData.name || '').split(' ')[0];

  const verifiedDot = document.getElementById('verified-dot');
  if (verifiedDot) verifiedDot.style.display = user.emailVerified ? 'block' : 'none';

  // Populate profile page
  document.getElementById('profile-name').textContent     = userData.name || '—';
  document.getElementById('profile-email').textContent    = user.email || '—';

  const verEl = document.getElementById('profile-verified');
  verEl.textContent = user.emailVerified ? '✓ Verified' : '✗ Not verified';
  verEl.className   = 'aval ' + (user.emailVerified ? 'verified' : 'unverified');

  // Signup timestamp (server-set, may be null on first load — use creationTime fallback)
  const createdAt = user.metadata.creationTime;
  document.getElementById('profile-joined').textContent = createdAt ? new Date(createdAt).toLocaleDateString() : '—';

  // Last login
  const lastLogin = user.metadata.lastSignInTime;
  document.getElementById('profile-lastlogin').textContent = lastLogin ? new Date(lastLogin).toLocaleString() : '—';

  // Restore goal/level selectors
  selectedGoal   = userData.goal        || 'weight_loss';
  selectedFLevel = userData.fitnessLevel|| 'beginner';
  restoreSelector('#goal-opts',  selectedGoal,   'goal');
  restoreSelector('#level-opts', selectedFLevel, 'flevel');

  // Restore reminder toggles
  if (userData.reminders?.workout) applyToggle('workout', true);
  if (userData.reminders?.streak)  applyToggle('streak', true);

  buildExerciseGrid();
  updateChallenge();
  refreshUI();
  startStepCounter();

  showScreen('app');
  showPage('home');
}

function refreshUI() {
  if (!userData) return;
  const u = userData;

  document.getElementById('stat-streak').textContent   = u.streak + '🔥';
  document.getElementById('stat-workouts').textContent = u.workouts;
  document.getElementById('stat-points').textContent   = u.points;
  document.getElementById('home-name').innerHTML        =
    (u.name||'Champion').toUpperCase().split(' ')[0] + ' <span>💪</span>';

  // Level
  let lvIdx = 0;
  for (let i = LEVELS.length-1; i >= 0; i--) { if (u.points >= LEVELS[i].xp) { lvIdx = i; break; } }
  const nextXP = LEVELS[lvIdx+1] ? LEVELS[lvIdx+1].xp : LEVELS[lvIdx].xp;
  const prevXP = LEVELS[lvIdx].xp;
  const pct    = lvIdx+1 < LEVELS.length ? Math.round(((u.points - prevXP)/(nextXP - prevXP))*100) : 100;
  document.getElementById('level-name').textContent   = 'Level '+(lvIdx+1)+' — '+LEVELS[lvIdx].name;
  document.getElementById('level-xp').textContent     = u.points+' / '+nextXP+' XP';
  document.getElementById('level-fill').style.width   = pct+'%';

  // Progress page
  document.getElementById('prog-total').textContent       = u.workouts;
  document.getElementById('prog-streak-disp').textContent = '🔥 '+u.streak+' day streak';
  document.getElementById('prog-best-streak').textContent = u.bestStreak;
  document.getElementById('prog-total-reps').textContent  = u.totalReps;
  document.getElementById('prog-calories').textContent    = Math.round(u.calories);

  buildWeekChart(u);
  buildBreakdown(u);
  updateStepUI(u.steps || 0);
  buildBadges(u);
}

function buildWeekChart(u) {
  const days     = ['Mo','Tu','We','Th','Fr','Sa','Su'];
  const todayIdx = (new Date().getDay() + 6) % 7;
  const data     = u.weeklyData || [0,0,0,0,0,0,0];
  const max      = Math.max(...data, 1);
  document.getElementById('week-chart').innerHTML = days.map((d,i) => {
    const h = Math.round((data[i]/max)*70);
    return `<div class="chart-bar-wrap">
      <div class="chart-bar${i===todayIdx?' today':''}" style="height:${h}px"></div>
      <div class="chart-label">${d}</div>
    </div>`;
  }).join('');
}

function buildBreakdown(u) {
  const ec = u.exerciseCounts || {};
  const el = document.getElementById('exercise-breakdown');
  if (!Object.keys(ec).length) {
    el.innerHTML = '<p style="font-size:0.8rem;color:var(--muted)">No exercises yet. Start a workout!</p>';
    return;
  }
  el.innerHTML = Object.entries(ec).sort((a,b)=>b[1]-a[1]).map(([id,reps]) => {
    const ex = EXERCISES.find(e => e.id===id);
    return `<div style="display:flex;justify-content:space-between;align-items:center;padding:0.6rem 0.85rem;background:var(--card);border:1px solid var(--border);border-radius:10px;margin-bottom:0.5rem;">
      <span style="font-size:0.9rem;">${ex ? ex.icon : ''} ${ex ? ex.name : id}</span>
      <span style="font-family:var(--font-mono);font-size:0.8rem;color:var(--accent2);">${reps} reps</span>
    </div>`;
  }).join('');
}

// ════════════════════════════════════════
//  REP COUNTER (with anti-cheat)
// ════════════════════════════════════════
const MAX_REPS_PER_2S = 6; // 3 per second × 2s window

function addRep() {
  if (!timerRunning)    { showToast('Start the timer first!'); return; }
  if (!sessionExercise) { showToast('Select an exercise first!'); return; }

  const now = Date.now();
  repSpeedBucket = repSpeedBucket.filter(t => now - t < 2000);
  repSpeedBucket.push(now);

  if (repSpeedBucket.length > MAX_REPS_PER_2S) {
    document.getElementById('anticheat-warn').classList.add('show');
    return;
  }

  document.getElementById('anticheat-warn').classList.remove('show');
  sessionReps++;
  document.getElementById('rep-count-w').textContent = sessionReps;
}

function resetReps() {
  sessionReps = 0;
  repSpeedBucket = [];
  document.getElementById('rep-count-w').textContent = 0;
  document.getElementById('anticheat-warn').classList.remove('show');
}

// ════════════════════════════════════════
//  END WORKOUT → SAVE TO FIRESTORE
// ════════════════════════════════════════
async function endWorkout() {
  const MIN_SECONDS = 30;
  if (timerRunning && timerSeconds < MIN_SECONDS) {
    showToast('⚠️ Workout too short to save (min 30s)');
    return;
  }

  stopTimer(false);

  const u         = userData;
  const today     = getToday();
  const todayIdx  = (new Date().getDay() + 6) % 7;
  const yesterday = new Date(Date.now()-86400000).toISOString().split('T')[0];

  // Streak logic
  if (u.lastWorkoutDate === today) {
    // Already saved a workout today — streak stays
  } else if (u.lastWorkoutDate === yesterday) {
    u.streak++;
  } else {
    u.streak = 1;
  }

  u.lastWorkoutDate = today;
  if (u.streak > u.bestStreak) u.bestStreak = u.streak;

  // Stats
  u.workouts++;
  const timeMin    = timerSeconds / 60;
  const ex         = EXERCISES.find(e => e.id===sessionExercise);
  const calPerRep  = ex ? ex.cal : 0.4;
  const calBurned  = Math.round(sessionReps * calPerRep + timeMin * 3);
  u.calories      += calBurned;
  u.totalReps     += sessionReps;

  if (sessionExercise) {
    u.exerciseCounts = u.exerciseCounts || {};
    u.exerciseCounts[sessionExercise] = (u.exerciseCounts[sessionExercise]||0) + sessionReps;
  }

  if (!u.weeklyData || u.weeklyData.length < 7) u.weeklyData = [0,0,0,0,0,0,0];
  u.weeklyData[todayIdx]++;

  // XP
  const xp = Math.max(10, Math.round(sessionReps * 0.5 + timeMin * 5));
  u.points += xp;

  // Persist to Firestore immediately (don't debounce post-workout)
  clearTimeout(saveDebounce);
  await persistUserData();

  // Show summary modal
  const mm = String(Math.floor(timerSeconds/60)).padStart(2,'0');
  const ss = String(timerSeconds%60).padStart(2,'0');
  document.getElementById('sum-time').textContent = mm+':'+ss;
  document.getElementById('sum-reps').textContent = sessionReps;
  document.getElementById('sum-cal').textContent  = calBurned;
  document.getElementById('sum-xp').textContent   = '+'+xp+' XP Earned!';
  document.getElementById('summary-modal').classList.add('open');

  // Reset session state
  timerSeconds   = 0;
  sessionReps    = 0;
  timerRunning   = false;
  document.getElementById('workout-timer').textContent = '00:00';
  document.getElementById('rep-count-w').textContent   = 0;
  document.getElementById('timer-start-btn').textContent = '▶ Start';
  document.getElementById('timer-status').textContent    = 'Ready to start';

  refreshUI();
}

  document.getElementById('summary-modal').classList.remove('open');
  showPage('home');
}

}

function updateStepUI(steps) {
  document.getElementById('step-count').textContent = steps.toLocaleString();
  const pct  = Math.min(100, Math.round((steps/10000)*100));
  document.getElementById('step-fill').style.width = pct+'%';
  const msgs = ['Walk around to count steps','Keep going!','Halfway there!','Almost at goal!','Goal reached! 🎉'];
  const idx  = pct>=100?4:pct>=75?3:pct>=50?2:pct>=20?1:0;
  document.getElementById('step-msg').textContent = msgs[idx]+' '+pct+'%';
}

function startStepCounter() {
  const statusEl = document.getElementById('step-status');
  updateStepUI(userData.steps || 0);

  if (window.DeviceMotionEvent) {
    let stepThreshold = 12;
    let lastStepTime  = 0;
    let accelHistory  = [];

    const handleMotion = (e) => {
      const acc = e.accelerationIncludingGravity;
      if (!acc || acc.x===null) return;
      const mag  = Math.sqrt(acc.x**2 + acc.y**2 + acc.z**2);
      accelHistory.push(mag);
      if (accelHistory.length > 20) accelHistory.shift();
      const avg   = accelHistory.reduce((a,b)=>a+b,0)/accelHistory.length;
      const delta = Math.abs(mag - avg);
      const now   = Date.now();
      if (delta > stepThreshold && now - lastStepTime > 300) {
        lastStepTime = now;
        if (!userData) return;
        userData.steps = (userData.steps||0) + 1;
        updateStepUI(userData.steps);
        scheduleSave();
        statusEl.textContent = 'Counting...';
      }
    };

    if (typeof DeviceMotionEvent.requestPermission === 'function') {
      statusEl.textContent = 'Tap to enable';
      statusEl.style.cursor = 'pointer';
      statusEl.onclick = () => {
        DeviceMotionEvent.requestPermission().then(state => {
          if (state === 'granted') {
            window.addEventListener('devicemotion', handleMotion);
            statusEl.textContent = 'Active';
            statusEl.onclick = null;
          }
        }).catch(() => { statusEl.textContent = 'Denied'; });
      };
    } else {
      window.addEventListener('devicemotion', handleMotion);
      statusEl.textContent = 'Active';
    }
  } else {
    // Desktop simulation (for development testing)
    statusEl.textContent = 'Simulated';
    const simId = setInterval(() => {
      if (!userData || !document.getElementById('screen-app').classList.contains('active')) {
        clearInterval(simId); return;
      }
      userData.steps = (userData.steps||0) + Math.floor(Math.random()*3);
      updateStepUI(userData.steps);
      scheduleSave();
    }, 3500);
  }
}

function applyToggle(type, on) {
  const track = document.getElementById('rem-'+type+'-track');
  const thumb = document.getElementById('rem-'+type+'-thumb');
  const cb    = document.getElementById('rem-'+type);
  if (!track||!thumb||!cb) return;
  cb.checked = on;
  if (on) {
    track.style.background    = 'rgba(232,255,71,0.2)';
    track.style.borderColor   = 'var(--accent)';
    thumb.style.background    = 'var(--accent)';
    thumb.style.transform     = 'translateX(20px)';
  } else {
    track.style.background    = 'var(--bg3)';
    track.style.borderColor   = 'var(--border2)';
    thumb.style.background    = 'var(--muted)';
    thumb.style.transform     = 'translateX(0)';
  }
}
